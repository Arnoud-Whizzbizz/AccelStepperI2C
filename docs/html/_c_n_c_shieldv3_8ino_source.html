62 void setup() {
63 
64  // Important: initialize Wire before creating AccelStepperI2C objects
65  Wire.begin();
66 
67  // If the slave's and master's reset is not synchronized by hardware, after a master's reset the slave might
68  // think the master wants another stepper, not a first one, and will run out of steppers, sooner or later.
69  resetAccelStepperSlave(addr);
70 
71  Serial.begin(115200);
72  Serial.print("\n\nAccelStepperI2C demo - CNC Shield V3.00\n\n\n");
73 
74  X = new AccelStepperI2C(addr, AccelStepper::DRIVER, /* step pin */ 2, /* dir pin */ 5);
75  Y = new AccelStepperI2C(addr, AccelStepper::DRIVER, 3, 6, 0, 0, true);
76  Z = new AccelStepperI2C(addr, AccelStepper::DRIVER, 4, 7, 0, 0, true);
77  // you could add stepper A on pins 12 (step) and 13 (dir) after installing
78  // the two jumpers directly above the power terminal
79 
80  if ((X->myNum == -1) or (Y->myNum == -1) or (Z->myNum == -1)) {
81  Serial.print("Error: Could not add one or more steppers. Halting\n\n");
82  while (true) {}
83  }
84  Serial.print("Steppers added with internal numbers X = ");
85  Serial.print(X->myNum); Serial.print(", Y = ");
86  Serial.print(Y->myNum); Serial.print(", Z = ");
87  Serial.print(Z->myNum); Serial.print(".\n\n");
88 
89  /* The CNC shield by default pulls the four drivers' _EN_ pins high, making them inactive.
90  You can enable theim either by installing a jumper next to the reset button between EN and GND
91  *or* by software. As all four EN are connected to Arduino pin 8, we have to tell Accelstepper
92  about that pin and to call enableOutputs(). Before that, we have to tell it that _EN_ is active low
93  with setPinsInverted().
94  */
95  X->setEnablePin(8);
96  X->setPinsInverted(false, false, true); // directionInvert, stepInvert, enableInvert
97  X->enableOutputs();
98  Y->setEnablePin(8);
99  Y->setPinsInverted(false, false, true);
100  Y->enableOutputs();
101  Z->setEnablePin(8);
102  Z->setPinsInverted(false, false, true);
103  Z->enableOutputs();
104 
105  // tailor MaxSpeeds and acceleration to your steppers, here X is a Nema-14, Y and Z are geared 28BYJ-48's
106  X->setMaxSpeed(2000); Y->setMaxSpeed(500); Z->setMaxSpeed(500);
107  X->setAcceleration(200); Y->setAcceleration(300); Z->setAcceleration(300);
108 
109  // install endstop switch
110  X->setEndstopPin(endstopXpin, true, true); // activeLow works nicely together with internal pullup
111  X->setEndstopPin(10, true, true); // activeLow works nicely together with internal pullup
112  X->enableEndstops(); // true by default
113 
114  X->moveTo(5000); Y->moveTo(8000); Z->moveTo(12000); // set targets
115  X->runState(); Y->runState(); Z->runState(); // start the three state machines
116

129 void loop() {
130  Serial.print("X endstops="); Serial.print(X->endstops()); Serial.print(" - "); printStats(X); //Serial.print("Y - "); printStats(Y); Serial.print("Z - "); printStats(Z);
131  delay (1000);
132  if (abs(X->distanceToGo()) == 0) {
133  X->moveTo(0 - X->currentPosition());
134  // after the target has been reached, the state machine resets itself to state_stopped. So we must start it again.
135  X->runState();
136  }
137 
138  if (abs(Y->distanceToGo()) == 0) {
139  Y->moveTo(0 - Y->currentPosition()); Y->runState();
140  }
141 
142  if (abs(Z->distanceToGo()) == 0) {
143  Z->moveTo(0 - Z->currentPosition()); Z->runState();
144  }
145 
146 }
